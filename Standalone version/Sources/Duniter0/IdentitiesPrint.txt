MODULE Duniter0IdentitiesPrint;
	
	

	IMPORT
		
		B := Duniter0Blockchain, BA := Duniter0Basic, Dialog, TextMappers, TextModels, TextViews, Views;
	
	PROCEDURE Revoked*;
		
		VAR
			
			uid: B.String;
			pubkey: B.Pubkey;
			hash: B.Hash;
			bnb: INTEGER;
			ok, b, member: BOOLEAN;
			n: INTEGER;
			mt, t, exp: LONGINT;
			s: BA.DateTime;
			ss: Views.Title;
			tt: TextModels.Model;
			f: TextMappers.Formatter;
		
		BEGIN (*Revoked*)
			tt := TextModels.dir.New();
			f.ConnectTo(tt);
			ok := B.IdNextPubkey(TRUE, pubkey);
			WHILE ok DO
				b := B.IdPubComplete(pubkey, uid, member, hash, bnb, exp); ASSERT(b, 100);
				IF exp = BA.revoked THEN
					ASSERT(~member, 101);
					f.WriteString(pubkey); f.WriteTab;
					f.WriteString(uid); f.WriteLn; f.WriteTab;
					b := B.TimeOf(bnb, mt, t); ASSERT(b, 102);
					BA.TimestampToString(mt, s);
					f.WriteString(s);
					f.WriteLn;
				END;
				ok := B.IdNextPubkey(FALSE, pubkey);
			END;
			f.WriteLn;
			n := 0;
			ok := B.IdNextUid(TRUE, uid);
			WHILE ok DO
				b := B.IdUidComplete(uid, pubkey, member, hash, bnb, exp); ASSERT(b, 103);
				IF exp = BA.revoked THEN
					ASSERT(~member, 104);
					f.WriteString(uid); f.WriteTab;
					INC(n);
					f.WriteString(pubkey); f.WriteLn; f.WriteTab;
					b := B.TimeOf(bnb, mt, t); ASSERT(b, 105);
					BA.TimestampToString(mt, s);
					f.WriteString(s);
					f.WriteLn;
				END;
				ok := B.IdNextUid(FALSE, uid);
			END;
			f.WriteLn;
			f.WriteMsg("#Duniter0:RevokedNb");
			f.WriteString(" = ");
			f.WriteInt(n);
			f.WriteLn;
			Dialog.MapString("#Duniter0:RevokedM", ss);
			Views.OpenAux(TextViews.dir.New(tt), ss);
		END Revoked;

	PROCEDURE Members*;
		
		VAR
			
			uid: B.String;
			pubkey: B.Pubkey;
			hash: B.Hash;
			bnb: INTEGER;
			ok, b, member: BOOLEAN;
			n: INTEGER;
			mt, t, exp: LONGINT;
			s: BA.DateTime;
			ss: Views.Title;
			tt: TextModels.Model;
			f: TextMappers.Formatter;
		
		BEGIN (*Members*)
			tt := TextModels.dir.New();
			f.ConnectTo(tt);
			ok := B.IdNextPubkeyM(TRUE, pubkey);
			WHILE ok DO
				f.WriteString(pubkey); f.WriteTab;
				b := B.IdPubComplete(pubkey, uid, member, hash, bnb, exp); ASSERT(b, 100);
				f.WriteString(uid); f.WriteLn; f.WriteTab;
				b := B.TimeOf(bnb, mt, t); ASSERT(b, 101);
				BA.TimestampToString(mt, s);
				f.WriteString(s);
				f.WriteString(" → ");
				BA.TimestampToString(exp, s);
				f.WriteString(s);
				f.WriteLn;
				ok := B.IdNextPubkeyM(FALSE, pubkey);
			END;
			f.WriteLn;
			n := 0;
			ok := B.IdNextUidM(TRUE, uid);
			WHILE ok DO
				f.WriteString(uid); f.WriteTab;
				b := B.IdUidComplete(uid, pubkey, member, hash, bnb, exp); ASSERT(b, 102);
				INC(n);
				f.WriteString(pubkey); f.WriteLn; f.WriteTab;
				b := B.TimeOf(bnb, mt, t); ASSERT(b, 103);
				BA.TimestampToString(mt, s);
				f.WriteString(s);
				f.WriteString(" → ");
				BA.TimestampToString(exp, s);
				f.WriteString(s);
				f.WriteLn;
				ok := B.IdNextUidM(FALSE, uid);
			END;
			f.WriteLn;
			ASSERT(n = B.IdLenM(), 60);
			f.WriteMsg("#Duniter0:MembersNb");
			f.WriteString(" = ");
			f.WriteInt(n);
			f.WriteLn;
			Dialog.MapString("#Duniter0:Identities", ss);
			Views.OpenAux(TextViews.dir.New(tt), ss);
		END Members;
	
	END Duniter0IdentitiesPrint.
