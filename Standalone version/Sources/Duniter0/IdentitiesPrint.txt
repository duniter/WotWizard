MODULE Duniter0IdentitiesPrint;
	
	

	IMPORT
		
		B := Duniter0Blockchain, BA := Duniter0Basic, Dialog, TextMappers, TextModels, TextViews, Views;
	
	PROCEDURE Print*;
		
		VAR
			
			uid: B.String;
			pubkey: B.Pubkey;
			hash: B.Hash;
			bnb: INTEGER;
			ok, b, member: BOOLEAN;
			n: INTEGER;
			mt, t, exp: LONGINT;
			s: BA.DateTime;
			ss: Views.Title;
			tt: TextModels.Model;
			f: TextMappers.Formatter;
		
		BEGIN (*Print*)
			tt := TextModels.dir.New();
			f.ConnectTo(tt);
			ok := B.IdNextPubkeyM(TRUE, pubkey);
			WHILE ok DO
				f.WriteString(pubkey); f.WriteTab;
				b := B.IdPubComplete(pubkey, uid, member, hash, bnb, exp); ASSERT(b);
				f.WriteString(uid); f.WriteLn; f.WriteTab;
				b := B.TimeOf(bnb, mt, t); ASSERT(b);
				BA.TimestampToString(mt, s);
				f.WriteString(s);
				f.WriteString(" → ");
				BA.TimestampToString(exp, s);
				f.WriteString(s);
				f.WriteLn;
				ok := B.IdNextPubkeyM(FALSE, pubkey);
			END;
			f.WriteLn;
			n := 0;
			ok := B.IdNextUidM(TRUE, uid);
			WHILE ok DO
				f.WriteString(uid); f.WriteTab;
				b := B.IdUidComplete(uid, pubkey, member, hash, bnb, exp); ASSERT(b);
				IF member THEN
					INC(n);
				END;
				f.WriteString(pubkey); f.WriteLn; f.WriteTab;
				b := B.TimeOf(bnb, mt, t); ASSERT(b);
				BA.TimestampToString(mt, s);
				f.WriteString(s);
				f.WriteString(" → ");
				BA.TimestampToString(exp, s);
				f.WriteString(s);
				f.WriteLn;
				ok := B.IdNextUidM(FALSE, uid);
			END;
			f.WriteLn;
			ASSERT(n = B.IdLenM());
			f.WriteMsg("#Duniter0:MembersNb");
			f.WriteString(" = ");
			f.WriteInt(n);
			f.WriteLn;
			Dialog.MapString("#Duniter0:Identities", ss);
			Views.OpenAux(TextViews.dir.New(tt), ss);
		END Print;
	
	END Duniter0IdentitiesPrint.

Duniter0IdentitiesPrint.Print;
