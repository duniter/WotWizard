MODULE Duniter0CentralitiesStressDJson;
	
	
	
	IMPORT
		
		B := Duniter0Blockchain, C := Duniter0CentralitiesStressD, J := UtilJson, Dialog, TextMappers, TextModels, TextViews, Views;
	
	PROCEDURE Print*;
		
		VAR
			
			centers, centersId: C.Centrals;
			i: INTEGER;
			s: Views.Title;
			t: TextModels.Model;
			f: TextMappers.Formatter;
		
		BEGIN (*Print*)
			C.Count(centers, centersId);
			J.StartObject;
			J.PushString("Centralities");
			J.BuildField("description");
			J.PushInteger(B.LastBlock());
			J.BuildField("block");
			J.StartArray;
			FOR i := 0 TO LEN(centersId) - 1 DO
				J.StartObject;
				J.PushString(centersId[i].id);
				J.BuildField("id");
				J.PushReal(100. * centersId[i].c);
				J.BuildField("centrality");
				J.BuildObject;
			END;
			J.BuildArray;
			J.BuildField("values");
			J.BuildObject;
			t := TextModels.dir.New();
			f.ConnectTo(t);
			J.GetJson().Write(f);
			Dialog.MapString("#Duniter0:Centralities", s);
			Views.OpenAux(TextViews.dir.New(t), s);
		END Print;
		
	END Duniter0CentralitiesStressDJson.

DevDebug.UnloadThis Duniter0CentralitiesStressDPrint Duniter0CentralitiesStressDJson Duniter0CentralitiesStressD 

Duniter0CentralitiesStressDJson.Print;
