MODULE Duniter0MembersPrint;
	
	

	IMPORT
		
		M := Duniter0Members, TextMappers, TextModels, TextViews, Views;

	PROCEDURE Do*;
		
		
		VAR
			
			t: TextModels.Model;
			f: TextMappers.Formatter;
			a: POINTER TO ARRAY OF INTEGER;
			i: INTEGER;
		
		BEGIN (*Do*)
			M.Count(a);
			t := TextModels.dir.New();
			f.ConnectTo(t);
			FOR i := 0 TO LEN(a) - 1 DO
				f.WriteTab;
				f.WriteInt(i);
				f.WriteTab;
				f.WriteInt(a[i]);
				f.WriteLn;
			END;
			Views.OpenView(TextViews.dir.New(t));
		END Do;
	
	END Duniter0MembersPrint.

Duniter0MembersPrint.Do;
