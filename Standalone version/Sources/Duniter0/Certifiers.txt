MODULE Duniter0Certifiers;
	
	

	IMPORT
		
		B := Duniter0Blockchain, StdLog, UtilSortI;
	
	PROCEDURE Count* (OUT nbCertifs: POINTER TO ARRAY OF INTEGER; OUT totCertifs: INTEGER);
		
		VAR
			
			ok: BOOLEAN;
			pos: B.CertPos;
			i: INTEGER;
		
		BEGIN (*Count*)
			totCertifs := 0;
			NEW(nbCertifs, B.IdLen());
			i := 0;
			ok := B.CertNextFrom(TRUE, pos);
			WHILE ok DO
				nbCertifs[i] := pos.CertPosLen();
				INC(totCertifs, nbCertifs[i]);
				INC(i);
				ok := B.CertNextFrom(FALSE, pos);
			END;
			UtilSortI.QuickSort(nbCertifs);
		END Count;
	
	PROCEDURE Print*;
		
		VAR
			
			nbCertifs: POINTER TO ARRAY OF INTEGER;
			i, m, n, n50, n90, totCertifs: INTEGER;
		
		BEGIN (*Print*)
			StdLog.Clear;
			StdLog.Open;
			Count(nbCertifs, totCertifs);
			m := LEN(nbCertifs);
			FOR i := 0 TO m - 1 DO
				StdLog.Tab;
				StdLog.Int(i);
				StdLog.Tab;
				StdLog.Int(nbCertifs[i]);
				StdLog.Ln;
			END;
			StdLog.String('~'); StdLog.Ln;
			StdLog.String("Certificateur -> Nombre de certifications");
			StdLog.Ln; StdLog.Ln;
			n := 0;
			FOR i := 0 TO m - 1 DO
				INC(n, nbCertifs[i]);
				IF n < totCertifs * 0.5 THEN
					n50 := i;
				END;
				IF n < totCertifs * 0.1 THEN
					n90 := i;
				END;
			END;
			StdLog.String("Proportion des plus gros certificateurs faisant 50% ou plus des certifications : ");
			StdLog.Real((m - n50) / m); StdLog.Ln;
			StdLog.String("Proportion des plus gros certificateurs faisant 90% ou plus des certifications : ");
			StdLog.Real((m - n90) / m); StdLog.Ln;
		END Print;
	
	END Duniter0Certifiers.

Duniter0Certifiers.Print;
