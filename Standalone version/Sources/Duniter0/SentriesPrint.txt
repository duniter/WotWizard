MODULE Duniter0SentriesPrint;
	
	

	IMPORT
		
		A := UtilAvlTree, B := Duniter0Blockchain, BA := Duniter0Basic, Dialog, Math, StdLog;
	
	TYPE
		
		Uid = POINTER TO RECORD (A.Elem)
			uid: B.String;
		END;
	
	PROCEDURE (i1: Uid) Compare (i2: A.Elem): BYTE;
		
		BEGIN (*Compare*)
			WITH i2: Uid DO
				RETURN BA.CompP(i1.uid, i2.uid);
			END;
		END Compare;
	
	PROCEDURE Do*;
		
		VAR
			
			ok, b: BOOLEAN;
			pubkey: B.Pubkey;
			hash: B.Hash;
			bnb, n: INTEGER;
			ids: A.Tree;
			id: Uid;
			e: A.Elem;
			s: Dialog.String;
		
		BEGIN (*Do*)
			StdLog.Clear; StdLog.Open;
			Dialog.MapString("#Duniter0:Threshold", s);
			StdLog.String(s + " = ");
			StdLog.Int(SHORT(ENTIER(Math.Ceiling(Math.Power(B.IdLen(), 1 / B.pars.stepMax)))));
			StdLog.Ln; StdLog.Ln;
			A.New(ids);
			ok := B.NextSentry(TRUE, pubkey);
			WHILE ok DO
				NEW(id);
				b := B.IdPub(pubkey, id.uid, hash, bnb); ASSERT(b);
				e := id;
				b := ids.SearchIns(e, n); ASSERT(~ b);
				ok := B.NextSentry(FALSE, pubkey);
			END;
			e := ids.Next(NIL);
			WHILE e # NIL DO
				StdLog.String(e(Uid).uid); StdLog.Ln;
				e := ids.Next(e);
			END;
			StdLog.Ln;
			Dialog.MapString("#Duniter0:SentriesNb", s);
			StdLog.String(s + " = ");
			StdLog.Int(B.SentriesLen()); StdLog.Ln;
		END Do;
	
	END Duniter0SentriesPrint.Do;
