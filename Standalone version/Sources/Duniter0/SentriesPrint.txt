MODULE Duniter0SentriesPrint;
	
	

	IMPORT
		
		A := UtilAvlTree, B := Duniter0Blockchain, BA := Duniter0Basic, Math, Dialog, TextMappers, TextModels, TextViews, Views;
	
	TYPE
		
		Uid = POINTER TO RECORD (A.Elem)
			uid: B.String;
		END;
	
	PROCEDURE (i1: Uid) Compare (i2: A.Elem): BYTE;
		
		BEGIN (*Compare*)
			WITH i2: Uid DO
				RETURN BA.CompP(i1.uid, i2.uid);
			END;
		END Compare;
	
	PROCEDURE Do*;
		
		VAR
			
			ok, b: BOOLEAN;
			pubkey: B.Pubkey;
			n: INTEGER;
			ids: A.Tree;
			id: Uid;
			e: A.Elem;
			s: Views.Title;
			t: TextModels.Model;
			f: TextMappers.Formatter;
		
		BEGIN (*Do*)
			t := TextModels.dir.New();
			f.ConnectTo(t);
			f.WriteMsg("#Duniter0:Threshold");
			f.WriteString(" = ");
			f.WriteInt(SHORT(ENTIER(Math.Ceiling(Math.Power(B.IdLen(), 1 / B.pars.stepMax)))));
			f.WriteLn; f.WriteLn;
			A.New(ids);
			ok := B.NextSentry(TRUE, pubkey);
			WHILE ok DO
				NEW(id);
				b := B.IdPub(pubkey, id.uid); ASSERT(b);
				e := id;
				b := ids.SearchIns(e, n); ASSERT(~ b);
				ok := B.NextSentry(FALSE, pubkey);
			END;
			e := ids.Next(NIL);
			WHILE e # NIL DO
				f.WriteString(e(Uid).uid); f.WriteLn;
				e := ids.Next(e);
			END;
			f.WriteLn;
			f.WriteMsg("#Duniter0:SentriesNb");
			f.WriteString(" = ");
			f.WriteInt(B.SentriesLen()); f.WriteLn;
			Dialog.MapString("#Duniter0:Sentries", s);
			Views.OpenAux(TextViews.dir.New(t), s);
		END Do;
	
	END Duniter0SentriesPrint.Do;
