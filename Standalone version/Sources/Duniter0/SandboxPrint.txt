MODULE Duniter0SandboxPrint;
	
	

	IMPORT
		
		B := Duniter0Blockchain, BA := Duniter0Basic, S := Duniter0Sandbox, StdLog;
	
	PROCEDURE Print*;
		
		VAR
			
			hash, toHash: S.Hash;
			uid: S.String;
			pubkey, from, to: S.Pubkey;
			exp: LONGINT;
			ok, okP, b: BOOLEAN;
			pos: S.CertPos;
			s: BA.DateTime;
		
		BEGIN (*Print*)
			StdLog.Clear; StdLog.Open;
			ok := S.IdNextHash(TRUE, hash);
			WHILE ok DO
				b := S.IdHash(hash, pubkey, uid, exp); ASSERT(b);
				StdLog.String(hash); StdLog.Tab;
				StdLog.String(pubkey); StdLog.Tab;
				StdLog.String(uid); StdLog.Tab;
				BA.TimestampToString(exp, s);
				StdLog.String(s);
				StdLog.Ln;
				ok := S.IdNextHash(FALSE, hash);
			END;
			StdLog.Ln;
			ok := S.IdNextPubkey(TRUE, pubkey, hash);
			WHILE ok DO
				StdLog.String(pubkey); StdLog.Tab;
				StdLog.String(hash);
				StdLog.Ln;
				ok := S.IdNextPubkey(FALSE, pubkey, hash);
			END;
			StdLog.Ln;
			ok := S.IdNextUid(TRUE, uid, hash);
			WHILE ok DO
				StdLog.String(uid); StdLog.Tab;
				StdLog.String(hash);
				StdLog.Ln;
				ok := S.IdNextUid(FALSE, uid, hash);
			END;
			StdLog.Ln;
			
			ok := S.CertNextFrom(TRUE, pos);
			WHILE ok DO
				okP := pos.CertNextPos(from, toHash);
				WHILE okP DO
					StdLog.String(from);
					IF B.IdPub(from, uid) THEN
						StdLog.String(" ("); StdLog.String(uid); StdLog.String(")");
					END;
					StdLog.String(" -> ");
					StdLog.String(toHash);
					IF S.IdHash(toHash, pubkey, uid, exp) OR S.Cert(from, toHash, to, exp) & B.IdPub(to, uid) THEN
						StdLog.String(" ("); StdLog.String(uid); StdLog.String(")");
					END;
					b := S.Cert(from, toHash, to, exp); ASSERT(b);
					BA.TimestampToString(exp, s);
					StdLog.String(s);
					StdLog.Ln;
					okP := pos.CertNextPos(from, toHash);
				END;
				ok := S.CertNextFrom(FALSE, pos);
			END;
			StdLog.Ln;
			ok := S.CertNextTo(TRUE, pos);
			WHILE ok DO
				okP := pos.CertNextPos(from, toHash);
				WHILE okP DO
					StdLog.String(toHash);
					IF S.IdHash(toHash, to, uid, exp) OR S.Cert(from, toHash, to, exp) & B.IdPub(to, uid) THEN
						StdLog.String(" ("); StdLog.String(uid); StdLog.String(")");
					END;
					StdLog.String(" <- ");
					StdLog.String(from);
					IF B.IdPub(from, uid) THEN
						StdLog.String(" ("); StdLog.String(uid); StdLog.String(")");
					END;
					b := S.Cert(from, toHash, to, exp); ASSERT(b);
					BA.TimestampToString(exp, s);
					StdLog.String(s);
					StdLog.Ln;
					okP := pos.CertNextPos(from, toHash);
				END;
				ok := S.CertNextTo(FALSE, pos);
			END;
		END Print;
	
	END Duniter0SandboxPrint.

Duniter0SandboxPrint.Print;
