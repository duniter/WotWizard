MODULE DuniterSandboxPrint;
	
	

	IMPORT
		
		B := DuniterBlockchain, BA := DuniterBasic, S := DuniterSandbox, StdLog;
	
	PROCEDURE Print*;
		
		VAR
			
			hash, toHash: S.Hash;
			uid: S.String;
			pubkey, from, to: S.Pubkey;
			bnb: INTEGER;
			exp, exp2: LONGINT;
			ok, okP, b: BOOLEAN;
			pos: S.CertPos;
			s: BA.DateTime;
		
		BEGIN (*Print*)
			StdLog.Clear; StdLog.Open;
			ok := S.IdNextHash(TRUE, hash);
			WHILE ok DO
				b := S.IdHash(hash, pubkey, uid, exp); ASSERT(b);
				StdLog.String(pubkey); StdLog.Tab;
				StdLog.String(uid); StdLog.Tab;
				BA.TimestampToString(exp, s);
				StdLog.String(s);
				StdLog.Ln;
				ok := S.IdNextHash(FALSE, hash);
			END;
			StdLog.Ln;
			ok := S.IdNextPubkey(TRUE, pubkey);
			WHILE ok DO
				StdLog.String(pubkey); StdLog.Tab;
				b := S.IdPub(pubkey, hash); ASSERT(b);
				StdLog.String(hash);
				StdLog.Ln;
				ok := S.IdNextPubkey(FALSE, pubkey);
			END;
			StdLog.Ln;
			ok := S.IdNextUid(TRUE, uid);
			WHILE ok DO
				StdLog.String(uid); StdLog.Tab;
				b := S.IdUid(uid, hash); ASSERT(b);
				StdLog.String(hash);
				StdLog.Ln;
				ok := S.IdNextUid(FALSE, uid);
			END;
			StdLog.Ln;
			
			ok := S.CertNextFrom(TRUE, pos);
			WHILE ok DO
				okP := pos.CertNextPos(from, to, toHash, exp);
				WHILE okP DO
					StdLog.String(from);
					IF B.IdPub(from, uid, bnb) THEN
						StdLog.String(" ("); StdLog.String(uid); StdLog.String(")");
					END;
					StdLog.String(" -> ");
					StdLog.String(to);
					IF S.IdHash(toHash, pubkey, uid, exp2) OR B.IdPub(to, uid, bnb) THEN
						StdLog.String(" ("); StdLog.String(uid); StdLog.String(")");
					END;
					BA.TimestampToString(exp, s);
					StdLog.String(s);
					StdLog.Ln;
					okP := pos.CertNextPos(from, to, toHash, exp);
				END;
				ok := S.CertNextFrom(FALSE, pos);
			END;
			StdLog.Ln;
			ok := S.CertNextTo(TRUE, pos);
			WHILE ok DO
				okP := pos.CertNextPos(from, to, toHash, exp);
				WHILE okP DO
					StdLog.String(to);
					IF S.IdHash(toHash, to, uid, exp2) OR B.IdPub(to, uid, bnb) THEN
						StdLog.String(" ("); StdLog.String(uid); StdLog.String(")");
					END;
					StdLog.String(" <- ");
					StdLog.String(from);
					IF B.IdPub(from, uid, bnb) THEN
						StdLog.String(" ("); StdLog.String(uid); StdLog.String(")");
					END;
					BA.TimestampToString(exp, s);
					StdLog.String(s);
					StdLog.Ln;
					okP := pos.CertNextPos(from, to, toHash, exp);
				END;
				ok := S.CertNextTo(FALSE, pos);
			END;
		END Print;
	
	END DuniterSandboxPrint.

DuniterSandboxPrint.Print;
