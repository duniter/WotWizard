MODULE Duniter1GQLReceiver;
	
	

	IMPORT
		
		BA := Duniter1Basic, G := UtilGraphQL, Dialog, Strings;
	
	CONST
		
		queryDir = "Query";
		queryName = "Query.txt";
		
		waitDelay = 100;
	
	TYPE
		
		WaitFor = POINTER TO RECORD (BA.WaitFor)
		END;
	
	PROCEDURE Start*;
		
		VAR
			
			w: WaitFor;
		
		BEGIN (*Start*)
			BA.SwitchOff(queryDir, queryName);
			NEW(w);
			w.Do(queryDir, queryName, waitDelay);
		END Start;
	
	PROCEDURE (w: WaitFor) End (IN name: ARRAY OF CHAR);
		
		VAR
			
			sel: G.Selection;
			call: Dialog.String;
			res: INTEGER;
			v: G.Value;
			s: Dialog.String;
		
		BEGIN (*End*)
			sel := G.ReadGraphQL(queryDir, queryName); ASSERT(sel # NIL, 100);
			call := sel.alias + ".json";
			IF sel.selSet.arguments # NIL THEN
				v := sel.selSet.arguments.value;
				WITH
					|v: G.BooleanValue DO
						IF v.boolean THEN
							s := "TRUE";
						ELSE
							s := "FALSE";
						END;
					|v: G.EnumValue DO
						s := v.enum$;
					|v: G.FloatValue DO
						Strings.RealToString(v.float, s);
					|v: G.IntValue DO
						Strings.IntToString(v.int, s);
					|v: G.NullValue DO
						s := "NIL";
					|v: G.StringValue DO
						s := v.string$;
				END;
				call := call + "', '" + s$;
			END;
			call := sel.name$ + "." + sel.selSet.name$ + "('" + call + "')";
			Dialog.Call(call, "", res); ASSERT(res = 0, 101);
			Start;
		END End;
	
	END Duniter1GQLReceiver.
