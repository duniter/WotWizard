MODULE BabelInit;
	
	(* The module BabelInit is part of the Babel subsystem, a compiler compiler. BabelInit loads a hard coded version of the tbl file corresponding to the definition document BabelSrc.odc. *)
	
	
	
	IMPORT
		
		C := BabelCompil, Dialog, Kernel;
	
	TYPE
		
		Directory = POINTER TO RECORD (C.Directory)
			pos: INTEGER;
		END;
		
		pInt* = POINTER TO ARRAY OF INTEGER;
	
	VAR
		
		a*: pInt;
	
	PROCEDURE (d: Directory) ReadInt (OUT i: INTEGER);
		
		BEGIN (*ReadInt*)
			i := a[d.pos];
			INC(d.pos);
		END ReadInt;
	
	PROCEDURE InitBabel* (): C.Compiler;
		
		VAR
			
			d: Directory;
			res: INTEGER;
			c: C.Compiler;
		
		BEGIN (*InitBabel*)
			NEW(d);
			Dialog.Call("BabelInit0.RemplisA", "", res);
			ASSERT(res = 0);
			Kernel.UnloadMod(Kernel.ThisMod("BabelInit0"));
			d.pos := 0;
			c := d.ReadCompiler();
			a := NIL;
			RETURN c;
		END InitBabel;
	
	END BabelInit.
